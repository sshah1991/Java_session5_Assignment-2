package myPack;

import java.util.Scanner;

public class VariableStack implements interfaceStack {
	int value;
	int stackSize = 5;
	int arrStack[] = new int[stackSize];
	int count = 0;
	int ch = 0;
	Scanner input = new Scanner(System.in);

	public VariableStack() {
		while (ch != 3) {
			System.out.println("enter your choice");
			System.out.println("1.Push");
			System.out.println("2.Pop");
			System.out.println("3.Exit");
			ch = input.nextInt();
			switch (ch) {
			case 1:
				// push
				if (count < stackSize) {
					System.out.println("enter the integer value");
					value = input.nextInt();
					push(value);
					break;
				} else {
					System.out.println("incrementing size");
					stackSize++;
					push(value);
					break;
				}
			case 2:
				// pop
				if (count < 0) {
					System.out.println("no elements to delete");
				} else {
					pop();

					break;
				}
			default:
				break;

			}
		}
	}

	public void push(int i) {
		// System.out.println("inside push");
		arrStack[count] = i;
		// System.out.println("inside push1");
		count++;
		// System.out.println("inside push2");

	}

	public void pop() {
		count--;

	}

}
